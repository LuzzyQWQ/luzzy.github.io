保存缓存

使用getSharedPreferences() 来设置数据库字典

apply没有返回值而commit返回boolean表明修改是否提交成功

// 设置共享内存，上传
val preferences = getSharedPreferences("database", Context.MODE_PRIVATE)
preferences.edit().apply {
    putString("username",username.text.toString())
    putString("password",password.text.toString())
}.apply()

使用Retrofit + Gson 实现网络请求

注意几点

  • 如果用http协议，需要写一个xml来开启http
    <?xml version="1.0" encoding="utf-8"?>
<network-security-config>
    <base-config cleartextTrafficPermitted="true" />
</network-security-config>
  • 在AndroidManifest.xml中加入网络权限请求
<uses-permission android:name="android.permission.INTERNET"/>
<uses-permission android:name="android.permission.ACCESS_NETWORK_STATE"/>

先生成retrofit Builder

//其中url在本地时，需要用公网ip
val retrofit = Retrofit.Builder()
    .baseUrl("http://10.20.35.213:8333/")
    .addConverterFactory(GsonConverterFactory.create())
    .build()

接口设置

interface LoginGetAPI {
    // 代表使用Get方法，根目录下
    // 发送 name 和 password 
    // 返回的是一个JSON

    @GET("/")
    fun loginGet(
        @Query("name") name: String,
        @Query("password") password: String
    ):Call<String>
}
// 创建api
val api = retrofit.create(LoginGetAPI::class.java)

// 调用网络请求
// enqueue代表等待回传
api.loginGet("name", "hey")
    .enqueue(object : Callback<String> {
        //自动生成收到回应和失败
        override fun onResponse(call: Call<String>, response: Response<String>) {
            preferences.edit().apply {
                putString("username", "丁丁")
                putString("password", "丁丁的密码")
            }.apply()
            startActivity(Intent(intent, TargetPage::class.java))
            Log.d("Response", "complete!")
        }

        override fun onFailure(call: Call<String>, t: Throwable) {
            Log.d("Hey",t.toString())
        }
    })
